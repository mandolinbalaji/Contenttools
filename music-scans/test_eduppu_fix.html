function calculateLeadingCommas(noteCount, cycleLength, eduppu, nadai = 1) {
    const eduppuCell = (eduppu - 1) * nadai + 1;
    let result = (eduppuCell - 1 - noteCount) % cycleLength;
    return result < 0 ? result + cycleLength : result;
}
